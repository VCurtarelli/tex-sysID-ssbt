\DeclareDocumentCommand{\bv}{m O{n} g}{
	\@my{#2}{\symbfup{#1}}\IfValueT{#3}{_{#3}}
}
\DeclareDocumentCommand{\BV}{m g}{
    \symbfsfup{#1}\IfValueT{#2}{_{#2}}
}
\DeclareDocumentCommand{\newbv}{s o m}{
    \IfBooleanTF{#1}{
        \IfValueTF{#2}{
            \@namedef{bv#2}{\BV{#3}}
        }{
            \@namedef{bv#3}{\BV{#3}}
        }
    }{
        \IfValueTF{#2}{
            \@namedef{bv#2}{\bv{#3}}
        }{
            \@namedef{bv#3}{\bv{#3}}
        }
    }
}
\DeclareDocumentCommand{\newd}{o m}{
	\IfValueTF{#1}{
		\@namedef{d#1}{\dot{#2}}
	}{
		\@namedef{d#2}{\dot{#2}}
	}
}
